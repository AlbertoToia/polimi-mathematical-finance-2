\section*{Binomial Model}
\textbf{Pricing}
\[
\begin{cases}
q_1 + q_2 = 1, \\
q_1 S_0 u + q_2 S_0 d = S_0 (1+r_n),
\end{cases}
\]
Contingent claim at time 0:
\[ \Phi(S_0) = \frac{1}{(1+r)^n} \sum_{k=0}^n \binom{n}{k} q_u^k q_d^{n-k} \phi^{(k) u; (n-k) d} \]

\textbf{No-Arbitrage and Completeness}\\
$d < 1+r < u$
\begin{itemize}
    \item NA $\Rightarrow$ market complete
    \item Every contingent claim is replicable
    \item Replication price = risk-neutral price
\end{itemize}

\textbf{Replicating Portfolio (iff NA)}
\[
\begin{cases}
\phi_{t+1}^u = x_t(1+r) + y_t S_t u = V_{t+1}^u, \\
\phi_{t+1}^d = x_t(1+r) + y_t S_t d = V_{t+1}^d
\end{cases}
\]

\textbf{Arbitrage Strategy Construction} -- Arbitrage exists if $\exists h$ such that:
$V_0^h = 0, \quad \mathbb{P}(V_T^h \ge 0) = 1, \quad \mathbb{P}(V_T^h > 0) > 0$

\textbf{Cox-Ross-Rubinstein (CRR)} \\
$1+R = e^{r \Delta t}$ \\
$u = e^{\sigma \sqrt{\Delta t}}, \quad d = e^{-\sigma \sqrt{\Delta t}} = 1/u$ \\
$q_u = \frac{e^{r \Delta t} - d}{u - d}, \quad q_d = 1 - q_u$
\section*{Stochastic Calculus}
\textbf{Standard Brownian Motion ($W_t$) Properties}
\begin{itemize}
    \item Initial value: $W_0 = 0$
    \item Increments: $W_t - W_s \sim \N(0, t-s)$
    \item $\E[W_t] = 0$
    \item $\text{Var}(W_t) = t$
    \item $\text{Cov}(W_s, W_t) = \min(s, t)$
    \item ItĹŤ rules: $(dW_t)^2 = dt, \quad (dt)^2 = 0$
\end{itemize}

\textbf{Geometric Brownian Motion (GBM)} \\
SDE: $dX_t = \mu X_t dt + \sigma X_t dW_t$ \\
Solution: $X_t = X_0 e^{(\mu - \frac{1}{2}\sigma^2)t + \sigma W_t}$

\textbf{ItĹŤ Lemma}\\
$(Z_t)_{t \ge 0}$ defined as $Z_t = f(X_t, t)$ satisfies:
\begin{align*}
dZ_t &= \df{f}{t}(X_t, t) \, dt + \df{f}{x}(X_t, t) \, dX_t + \frac{1}{2} \frac{\partial^2 f}{\partial x^2}(X_t, t) \, \sigma^2(t) \, dt \\
&= \left( \df{f}{t} + \mu \df{f}{x} + \frac{\sigma^2}{2} \frac{\partial^2 f}{\partial x^2} \right)(X_t, t) \, dt + \sigma(t) \df{f}{x}(X_t, t) \, dW_t
\end{align*}

The stoch. proc. $(Z_t)_{t \ge 0}$ defined as $Z_t = f(X_t^1, \dots, X_t^n, t)$ satisfies:
\begin{align*}
dZ_t &= \df{f}{t} dt + \sum_{i=1}^n \df{f}{x_i} dX_t^i + \frac{1}{2} \sum_{i,j=1}^n \sigma^i \sigma^j \rho_{ij} \frac{\partial^2 f}{\partial x_i \partial x_j} dt
\end{align*}
One dimensional formula: \\
$dZ_t = \left( \mu f'(X_t) + \frac{\sigma^2}{2} f''(X_t) \right) \, dt + \sigma f'(X_t) \, dW_t$

\textbf{Correlated $d$-dimensional Brownian Motion}
\begin{itemize}
    \item $B_t = (B_t^1, \dots, B_t^n)$, $\E[B_t^i]=0$
    \item $\text{Cov}(B_t^i, B_t^j) = \rho_{ij} t$, $d\langle B^i, B^j \rangle_t = \rho_{ij} dt$
    \item Correlation construction (2D): $B_t^1 = \sqrt{1-\rho^2} W_t^1 + \rho W_t^2$, $B_t^2 = W_t^2$ with $W^1 \perp W^2$.
\end{itemize}

\textbf{Multidimensional ItĹŤ's Lemma:} For $f(t, X_t^1, X_t^2)$ and $dX_t^i = \mu^i dt + \sigma^i dB_t^i$:
$df = \left\{ \df{f}{t} + \sum_i \mu^i f_{x_i} + \frac{1}{2} \sum_{i,j} \rho_{ij} \sigma^i \sigma^j f_{x_i x_j} \right\} dt + \sum_i f_{x_i} \sigma^i dB_t^i$

\textbf{Martingale Condition}\\
$M_t$ is a martingale iff $\E[M_t \mid \mathcal{F}_s] = M_s$, $0 \le s < t$.
In differential form, the DRIFT term must vanish.
In particular, if $dZ_t = \Phi_t dW_t, \Phi \in \mathcal{M}^2$, then $(Z_t)$ is a martingale.

\section*{Black \& Scholes}
\textbf{Black-Scholes Setup}
\[
\begin{cases}
    dB_t = r B_t dt \\
    dS_t = S_t (\alpha dt + \sigma dW_t) \\
    S_t = S_0 \exp\{(\alpha - \frac{1}{2}\sigma^2)t + \sigma W_t\}
\end{cases}
\]
\textbf{Black-Scholes PDE}
\[
\begin{cases}
F_t + r S F_S + \frac{\sigma^2}{2} S^2 F_{SS} - r F = 0 \\
F(T, S) = \Phi(S_T)
\end{cases}
\]

\textbf{Vanilla Call}
\[ C(S_t, t) = S_t N(d_1) - K e^{-r(T-t)} N(d_2) \]
\[ d_1 = \frac{\ln(S_t/K) + (r + \frac{\sigma^2}{2})(T-t)}{\sigma\sqrt{T-t}} \]
\[ d_2 = d_1 - \sigma\sqrt{T-t} \]

\textbf{Discrete Dividends Call}
\[ C(S_0, 0) = (S_0 - D) N(\hat{d}_1) - K e^{-rT} N(\hat{d}_2) \]
\[ \hat{d}_1 = \frac{\ln((S_0-D)/K) + (r + \frac{1}{2}\sigma^2)T}{\sigma\sqrt{T}} \]
\[ \hat{d}_2 = \hat{d}_1 - \sigma\sqrt{T} \]

\textbf{Continuous Dividends Call}
\[ C(S_t, t) = S_t e^{-\delta(T-t)} N(d_1') - K e^{-r(T-t)} N(d_2') \]
\[ d_1' = \frac{\ln(S_t/K) + (r - \delta + \frac{\sigma^2}{2})(T-t)}{\sigma\sqrt{T-t}} \]
\[ d_2' = d_1' - \sigma\sqrt{T-t} \]

\textbf{Vanilla Put}
\[ P(S_t, t) = K e^{-r(T-t)} N(-d_2) - S_t N(-d_1) \]
(Use same $d_1, d_2$ as Vanilla Call)

\textbf{Discrete Dividends Put}
\[ P(S_0, 0) = K e^{-rT} N(-\hat{d}_2) - (S_0 - D) N(-\hat{d}_1) \]
(Use same $\hat{d}_1, \hat{d}_2$ as Discrete Call)

\textbf{Continuous Dividends Put}
\[ P(S_t, t) = K e^{-r(T-t)} N(-d_2') - S_t e^{-\delta(T-t)} N(-d_1') \]
(Use same $d_1', d_2'$ as Continuous Call)

\textbf{Dividends Definitions} \\
Continuous: $D = S_0(1-e^{-\delta T})$ \\
Discrete: $D = \sum_i d_i e^{-rT_i}$

\subsection*{Probabilities of ITM}
\[ \mathbb{P}(\text{Call ITM}) = \mathbb{P}(S_T > K) = 1 - \mathbb{P}\left(Z < \frac{\ln(K/S_0) - (\alpha - \frac{1}{2}\sigma^2)T}{\sigma\sqrt{T}}\right) \]
\[ \mathbb{Q}(\text{Call ITM}) = 1 - \mathbb{Q}\left(Z < \frac{\ln(K/S_0) - (r - \frac{1}{2}\sigma^2)T}{\sigma\sqrt{T}}\right) = N(d_2) \]

\textbf{European Call Greeks}
\[ \Delta_c = N(d_1), \quad \Gamma_c = \frac{\varphi(d_1)}{S\sigma\sqrt{T-t}} \]
\[ \rho_c = K(T-t)e^{-r(T-t)}N(d_2) \]
\[ \mathcal{V}_c = S \varphi(d_1) \sqrt{T-t} \]
\[ \Theta_c = -\frac{S\sigma\varphi(d_1)}{2\sqrt{T-t}} - rKe^{-r(T-t)}N(d_2) \]

\textbf{European Put Greeks}
\[ \Delta_p = N(d_1) - 1, \quad \Gamma_p = \Gamma_c \]
\[ \rho_p = -K(T-t)e^{-r(T-t)}N(-d_2) \]
\[ \mathcal{V}_p = \mathcal{V}_c \]
\[ \Theta_p = -\frac{S\sigma\varphi(d_1)}{2\sqrt{T-t}} + rKe^{-r(T-t)}N(-d_2) \]
\textbf{Normal Density}: $\varphi(x) = \frac{1}{\sqrt{2\pi}} e^{-x^2/2}$

\section*{Hedging \& Others}
\textbf{Put-Call Parity} \\
\begin{itemize}
    \item No dividends: $C_t - P_t = S_t - K e^{-r(T-t)}$ \\
    \item Discrete dividends: $C_t - P_t = S_t - D - K e^{-rT}$ \\
    \item Continuous dividends: $C_t - P_t = S_t e^{-q(T-t)} - K e^{-r(T-t)}$
\end{itemize}

\textbf{Hedging Portfolio}: $V = x S + y C_h$ \\
\textbf{Greeks for Stock}: \\
$\Delta_S = 1, \quad \Gamma_S = 0, \quad \mathcal{V}_S = 0, \quad \rho_S = 0, \quad \Theta_S = 0$.

\textbf{Option Spreads}
\begin{itemize}
    \item \textbf{Bull Spread}: buy $C(K_{\text{low}})$, sell $C(K_{\text{high}})$
    \item \textbf{Bear Spread}: sell $P_1$, buy $P_2 (K_1 < K_2)$
    \item \textbf{Butterfly}: buy $C(K_1)$, sell $2C(K_2)$, buy $C(K_3)$ ($K_2 \approx \text{avg}$)
\end{itemize}
\textbf{Combinations}
\begin{itemize}
    \item \textbf{Straddle}: $P+C$, $K_P = K_C$
    \item \textbf{Strangle}: $P+C$, $K_P < K_C$
    \item \textbf{Strip}: $2P+C$, \textbf{Strap}: $P+2C$
\end{itemize}

\textbf{Inequalities for Options}
\begin{itemize}
    \item $C_t \le S_t; \quad P_t \le K e^{-r(T-t)}$
    \item $C_t \ge S_t - K e^{-r(T-t)}; \quad P_t \ge K e^{-r(T-t)} - S_t$
    \item $S_t - K \le C_t^A - P_t^A \le S_t - K e^{-r(T-t)}$
\end{itemize}

\section*{Meta Theorem}
$R = \#$ BM, $N = \#$ Risky Assets. \\
\textbf{No Arbitrage} $\iff R \ge N$ \\
\textbf{Completeness} $\iff R \le N$ \\
If market is incomplete ($R > N$), choose $\theta \in \mathbb{R}^R$ such that: \\
$\sigma \cdot \underline{\theta} = r \cdot \underline{1} - \mathbf{\alpha}$ \\
NA holds if $\sigma$ has full rank (i.e. $R=N$ and $\sigma$ is invertible).

\section*{Exotic Options}
\begin{itemize}
    \item \textbf{Cash-or-Nothing}: Pays $K$ if $S_T > L$ (Call).\\ Price $C = K e^{-r(T-t)} N(d_2)$. \\
    Put ($S_T < L$): $P = K e^{-r(T-t)} N(-d_2)$.
    \item \textbf{Asset-or-Nothing}: Pays $S_T 1_{S_T > K}$ (Call).\\Price $C = S_t N(d_1)$. \\
    Put ($S_T < K$): $P = S_t N(-d_1)$.
    \item \textbf{Chooser}: Right to choose at $T_1 < T_2$ if Call or Put. \\
    $\max(C_{T_1}, P_{T_1}) = C_{T_1} + \Phi(S_{T_1})$ where $\Phi(x) = [K e^{-r(T_2-T_1)} - x]^+$ \\
    $CH_0 = \E^\Q[e^{-r T_1} (C_{T_1} + \Phi(S_{T_1}))]$ \\
    $CH_0 = C(S_0, K, T_2) + P(S_0, K e^{-r(T_2-T_1)}, T_1)$
    \item \textbf{Asian}: $\left(\frac{1}{T+1} \sum S_{t_i} - K\right)^+$
    \item \textbf{Lookback}: Floating Strike ($S_T - \min S_u$), Fixed Strike ($\max S_u - K$).
\end{itemize}
\textbf{Barrier Options}:
K strike, L barrier.\\
$a = (L/S_t)^{\frac{2r}{\sigma^2} - 1}, b = (L/S_t)^{\frac{2r}{\sigma^2} + 1}$. \\
$d_1 = \frac{1}{\sigma \sqrt{T-t}} \left[ \ln \frac{S_t}{K} + (r+ \sigma^2/2)(T-t) \right]$ \\
$d_2 = \frac{1}{\sigma \sqrt{T-t}} \left[ \ln \frac{S_t}{K} + (r- \sigma^2/2)(T-t) \right]$ \\
$d_3 = \frac{1}{\sigma \sqrt{T-t}} \left[ \ln \frac{S_t}{L} + (r+ \sigma^2/2)(T-t) \right]$ \\
$d_4 = \frac{1}{\sigma \sqrt{T-t}} \left[ \ln \frac{S_t}{L} + (r- \sigma^2/2)(T-t) \right]$ \\
$d_5 = \frac{1}{\sigma \sqrt{T-t}} \left[ \ln \frac{S_t}{L} - (r- \sigma^2/2)(T-t) \right]$ \\
$d_6 = \frac{1}{\sigma \sqrt{T-t}} \left[ \ln \frac{S_t}{L} - (r+ \sigma^2/2)(T-t) \right]$ \\
$d_7 = \frac{1}{\sigma \sqrt{T-t}} \left[ \ln \frac{S_t K}{L^2} - (r- \sigma^2/2)(T-t) \right]$ \\
$d_8 = \frac{1}{\sigma \sqrt{T-t}} \left[ \ln \frac{S_t K}{L^2} - (r+ \sigma^2/2)(T-t) \right]$ \\
\\
\textbf{In-Out Parity}: $F_{IN} + F_{OUT} = F_{Vanilla}$

\textbf{Down-and-In (DI) ($S_t > L$)}
\begin{itemize}
    \item \textbf{Call}: \\
    If $K > L$: $C_{DI} = (L/S_t)^{\frac{2r}{\sigma^2} - 1} C(L^2/S_t, t)$ \\
    If $K < L$: $C_{DI} = C(S_t, t) - C_{DO}$ (Use $C_{DO}$ below)
    \item \textbf{Put}: \\
    If $K > L$: $P_{DI} = P(S_t, t) - P_{DO}$ (Use $P_{DO}$ below) \\
    If $K < L$: $P_{DI} = (L/S_t)^{\frac{2r}{\sigma^2} - 1} P(L^2/S_t, t)$
\end{itemize}

\textbf{Down-and-Out (DO) ($S_t > L$)}
\begin{itemize}
    \item \textbf{Call}: \\
    If $K > L$: $C_{DO} = C(S_t, t) - C_{DI}$ \\
    If $K < L$: $C_{DO} = S_t [N(d_3) - b(1-N(d_6))] - K e^{-r(T-t)} [N(d_4) - a(1-N(d_5))]$
    \item \textbf{Put}: \\
    If $K > L$: $P_{DO} = K e^{-r(T-t)} [N(d_4) - N(d_2) - a(N(d_7) - N(d_5))] - S_t [N(d_3) - N(d_1) - b(N(d_8) - N(d_6))]$ \\
    If $K < L$: $P_{DO} = 0$ (Knocked out $S>L>K$ impossible for Put payoff)
\end{itemize}

\textbf{Up-and-In (UI) ($S_t < L$)}
\begin{itemize}
    \item \textbf{Call}: \\
    If $K < L$: $C_{UI} = C(S_t, t) - C_{UO}$ (Use $C_{UO}$ below) \\
    If $K > L$: $C_{UI} = (L/S_t)^{\frac{2r}{\sigma^2} - 1} C(L^2/S_t, t)$
    \item \textbf{Put}: \\
    If $K < L$: $P_{UI} = (L/S_t)^{\frac{2r}{\sigma^2} - 1} P(L^2/S_t, t)$ \\
    If $K > L$: $P_{UI} = P(S_t, t) - P_{UO}$ (Use $P_{UO}$ below)
\end{itemize}

\textbf{Up-and-Out (UO) ($S_t < L$)}
\begin{itemize}
    \item \textbf{Call}: \\
    If $K < L$: $C_{UO} = S_t [N(d_1) - N(d_3) - b(N(d_6) - N(d_8))] - K e^{-r(T-t)} [N(d_2) - N(d_4) - a(N(d_5) - N(d_7))]$ \\
    If $K > L$: $C_{UO} = 0$
    \item \textbf{Put}: \\
    If $K < L$: $P_{UO} = P(S_t, t) - P_{UI}$ \\
    If $K > L$: $P_{UO} = K e^{-r(T-t)} [N(d_6) - b(1-N(d_3))] - S_t [N(d_5) - a(1-N(d_4))]$
\end{itemize}

\textbf{Futures Options (Black 76)} $F_0$ is the current future/forward price.
\textbf{Call on Future/Forward}: $C = e^{-rT} [F_0 N(d_1) - K N(d_2)]$ \\
\textbf{Put on Future/Forward}: $P = e^{-rT} [K N(-d_2) - F_0 N(-d_1)]$ \\
$d_1 = \frac{\ln(F_0/K) + \frac{1}{2}\sigma^2 T}{\sigma\sqrt{T}}$; \quad $d_2 = d_1 - \sigma\sqrt{T}$

\section*{Interest Rates}
\textbf{Affine Term Structure (ATS)} \\
$p(t, T) = \exp\{A(t, T) - B(t, T)r_t\}$ \\
Requires $\mu(r, t) = \alpha(t)r + \beta(t)$ and $\omega(r, t) = \sqrt{\gamma(t)r + \delta(t)}$ \\
where $dr_t = \mu(r_t, t) dt + \omega(r_t, t)dW_t$. \\
System:
$\begin{cases} B_t + \alpha(t) B - \frac{1}{2} \gamma(t) B^2 = -1; \quad B(T, T)=0 \\ A_t - \beta(t) B + \frac{1}{2} \delta(t) B^2 = 0; \quad A(T, T)=0 \end{cases}$ \\
For a calibrated model: $p(t, T) = \exp\{-\int_t^T f^*(0, s) ds\}$

$Z^{HL}(t, T) = \exp(-A^{HL}(t,T) - B^{HL}(t,T)r_t)$ (Ho-Lee) \\
$Z^{HW}(t, T) = \exp(-A^{HW}(t,T) - B^{HW}(t,T)r_t)$ (Hull-White) \\

\textbf{Vasicek Model}
$\alpha(t)=-a, \beta(t)=ab, \gamma(t)=0, \delta(t)=\sigma^2$.\\
$dr_t = a(b-r_t)dt + \sigma dW_t$ \\
$B(t, T) = \frac{1-e^{-a(T-t)}}{a}$ \\
$A(t, T) = \frac{(B(t,T) - (T-t))(a^2b - \sigma^2/2)}{a^2} - \frac{\sigma^2 B^2(t,T)}{4a}$ \\
$ r_t = r_0 e^{-at} + b(1-e^{-at}) + \sigma \int_0^t e^{-a(t-s)} dW_s $ \\
$\E[r_t] = r_0 e^{-at} + b(1-e^{-at})$

\textbf{Ho-Lee Model}
$\alpha(t)=0, \beta(t)=\theta_t, \gamma(t)=0, \delta(t)=\sigma^2$.\\
$dr_t = \theta_t dt + \sigma dW_t$ \\
$B(t, T) = T-t$ \\
$A(t, T) = \int_t^T \theta_s (s-T) ds + \frac{\sigma^2 (T-t)^3}{6}$ \\
Theta formula: $\theta^{HL}_t = \df{f(0,t)}{t} + \sigma^2 t$

\textbf{Hull-White Model}
$\alpha(t)=-a, \beta(t)=\theta_t, \gamma(t)=0, \delta(t)=\sigma^2$.\\
$dr_t = (\theta_t - ar_t)dt + \sigma dW_t$ \\
$B(t, T) = \frac{1 - e^{-a(T-t)}}{a}$ \\
$A(t, T) = \int_t^T [\frac{1}{2}\sigma^2 B^2(s, T) - \theta_s B(s, T) ] ds$ \\
Theta formula: $\theta^{HW}_t = \df{f(0,t)}{t} + a f(0, t) + \frac{\sigma^2}{2a}(1-e^{-2at})$

\textbf{Option Pricing on Zero-Coupon Bonds} \\
Use the normalized price $p(t, T) = \exp\{A(t, T) - B(t, T)r_t\}$,\\
same for the strike $K$: ALWAYS divide for $N$.\\
$C(t, T, K, T^*) = N\left[p(t, T^*)N(d_1) - \frac{K}{N} p(t, T)N(d_2)\right]$ \\
$d_1 = \frac{1}{\sigma_p} \left[\ln\left(\frac{p(t, T^*)}{(K/N) p(t, T)}\right) + \frac{1}{2}\sigma_p^2\right]$, \quad $d_2 = d_1 - \sigma_p$ \\

\textbf{Option Pricing (Vasicek)} \\
$\sigma_p = \sigma \frac{1-e^{-a(T^*-T)}}{a} \sqrt{\frac{1-e^{-2aT}}{2a}}$

\textbf{Option Pricing (Hull-White)} \\
$\sigma_p = \sigma \frac{1-e^{-a(T^*-T)}}{a} \sqrt{\frac{1-e^{-2aT}}{2a}}$

\textbf{Option Pricing (Ho-Lee)} \\
$\sigma_p = \sigma (T^* - T) \sqrt{T}$

\section*{Libor \& Swap}
\textbf{Variables}:
\textbf{Variables}: \\
$L(0, T_{i-1}, T_i)$: Forward LIBOR rate at $t=0$. \\
$P(0, T_i)$: Zero-coupon bond price. \\
$\bar{\sigma}$: Forward Implied Volatility. \\
$N(\cdot)$: CDF.

\textbf{Black-76 Formula for a Caplet} \\
$\text{Cpl}(0) = M \cdot \alpha \cdot P(0, T_i) [L(0, T_{i-1}, T_i) N(d_1) - R N(d_2)]$ \\
Auxiliary: \\
$d_1 = \frac{\ln(L/R) + \frac{1}{2}\Sigma_i^2(0, T_{i-1})}{\Sigma_i(0, T_{i-1})}$ \\
$d_2 = d_1 - \Sigma_i$ \\
Volatility Scaling: $\Sigma_i(0, T_{i-1}) = \bar{\sigma}\sqrt{T_{i-1}}$

\textbf{Zero-Coupon Bond}: $P(0, T) = \frac{1}{1 + L_{\text{spot}} \cdot T}$ \\
\textbf{Forward Rate}: $L(0, T_{i-1}, T_i) = \frac{P(0, T_{i-1})}{P(0, T_i) - 1} \text{ scaled by } \frac{1}{\alpha}$
